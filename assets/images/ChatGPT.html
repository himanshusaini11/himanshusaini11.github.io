#!/bin/bash

# Script to set up a basic GitHub Pages portfolio structure

# Check if a directory name (username.github.io) is provided
if [ -z "$1" ]; then
  echo "Usage: $0 username.github.io"
  exit 1
fi

# Set up the root directory
ROOT_DIR="$1"

# Create the root directory
mkdir -p "$ROOT_DIR"

# Navigate into the root directory
cd "$ROOT_DIR" || exit

# Create main files
touch index.html
echo "title: Your Portfolio Site" > _config.yml
echo "# $ROOT_DIR" > README.md
echo "MIT License" > LICENSE

# Create assets directories
mkdir -p assets/css assets/js assets/images

# Create basic CSS and JS files
touch assets/css/styles.css
touch assets/js/scripts.js

# Create project folders with example structure
for i in {1..3}; do
  mkdir -p "project$i"
  touch "project$i/index.html"
  touch "project$i/styles.css"
done

# Print a message
echo "Portfolio structure created successfully in $ROOT_DIR!"
